Quickstart

A short example of Ako to get started with.

import {
  Application,
} from "https://deno.land/x/ako/mod.ts";

const app = new Application();
app.use((ctx) => {
  ctx.body = "Hello Ako!";
});

app.listen({ port: 3000 });
